33.3.3 示例初始化文件 ¶

以下是一个 inputrc 文件的示例。这展示了键绑定、变量赋值和条件语法。 

# 此文件控制使用 GNU Readline 库的程序的行输入编辑行为。现有的
# 程序包括 FTP、Bash 和 GDB。
#
# 你可以使用 C-x C-r 重新读取 inputrc 文件。
# 以 '#' 开头的行是注释。
#
# 首先，包含 /etc/Inputrc 中的任何系统范围的绑定和变量
# 赋值
$include /etc/Inputrc

#
# 为 emacs 模式设置各种绑定。

set editing-mode emacs 

$if mode=emacs

Meta-Control-h:	向后删除单词	函数名后的文本将被忽略

#
# 小键盘模式下的箭头键
#
#"\M-OD":        向后移动字符
#"\M-OC":        向前移动字符
#"\M-OA":        上一条历史记录
#"\M-OB":        下一条历史记录
#
# ANSI 模式下的箭头键
#
"\M-[D":        向后移动字符
"\M-[C":        向前移动字符
"\M-[A":        上一条历史记录
"\M-[B":        下一条历史记录
#
# 8 位小键盘模式下的箭头键
#
#"\M-\C-OD":       向后移动字符
#"\M-\C-OC":       向前移动字符
#"\M-\C-OA":       上一条历史记录
#"\M-\C-OB":       下一条历史记录
#
# 8 位 ANSI 模式下的箭头键
#
#"\M-\C-[D":       向后移动字符
#"\M-\C-[C":       向前移动字符
#"\M-\C-[A":       上一条历史记录
#"\M-\C-[B":       下一条历史记录

C-q: 插入转义字符

$endif

# 一种旧风格的绑定。这恰好是默认设置。
TAB: 补全

# 方便与 shell 交互的宏
$if Bash
# 编辑路径
"\C-xp": "PATH=${PATH}\e\C-e\C-a\ef\C-f"
# 准备输入一个带引号的单词 --
# 插入左右双引号
# 并移动到左引号之后
"\C-x\"": "\"\"\C-b"
# 插入一个反斜杠（测试序列和宏中的反斜杠转义）
"\C-x\\": "\\"
# 引用当前或前一个单词
"\C-xq": "\eb\"\ef\""
# 添加一个绑定来刷新行，该绑定未绑定
"\C-xr": 重绘当前行
# 编辑当前行上的变量。
"\M-\C-v": "\C-a\C-k$\C-y\M-\C-e\C-a\C-y="
$endif

# 如果可用，使用可见的响铃
set bell-style visible

# 读取时不将字符截断为 7 位
set input-meta on

# 允许插入 iso-latin1 字符，而不是
# 将其转换为前缀元序列
set convert-meta off

# 直接显示设置了第八位的字符
# 而不是作为元前缀字符
set output-meta on

# 如果一个单词有 150 个或更多可能的补全项，
# 询问用户是否想查看所有补全项
set completion-query-items 150

# 对于 FTP
$if Ftp
"\C-xg": "get \M-?"
"\C-xt": "put \M-?"
"\M-.": 粘贴上一个参数
$endif
