33.3.3示例初始化文件 ¶

这是一个inputrc文件的示例。它展示了按键绑定、变量赋值和条件语法。 

# 此文件控制使用GNU Readline库进行行输入编辑的程序的行为。现有程序包括FTP、Bash和GDB。
#
# 你可以使用C-x C-r重新读取inputrc文件。
# 以'#'开头的行是注释。
#
# 首先，包含来自 /etc/Inputrc的任何系统范围的绑定和变量赋值
$include /etc/Inputrc

#
# 设置emacs模式的各种绑定。

set editing-mode emacs 

$if mode=emacs

Meta-Control-h:	backward-kill-word	函数名后面的数据将被忽略

#
# 小键盘模式下的箭头键
#
#"\M-OD":        backward-char
#"\M-OC":        forward-char
#"\M-OA":        previous-history
#"\M-OB":        next-history
#
# ANSI模式下的箭头键
#
"\M-[D":        backward-char
"\M-[C":        forward-char
"\M-[A":        previous-history
"\M-[B":        next-history
#
# 8位小键盘模式下的箭头键
#
#"\M-\C-OD":       backward-char
#"\M-\C-OC":       forward-char
#"\M-\C-OA":       previous-history
#"\M-\C-OB":       next-history
#
# 8位ANSI模式下的箭头键
#
#"\M-\C-[D":       backward-char
#"\M-\C-[C":       forward-char
#"\M-\C-[A":       previous-history
#"\M-\C-[B":       next-history

C-q: quoted-insert

$endif

# 一个老式绑定。这恰好是默认设置。
TAB: complete

# 方便与shell交互的宏
$if Bash
# 编辑路径
"\C-xp": "PATH=${PATH}\e\C-e\C-a\ef\C-f"
# 准备输入一个带引号的单词 --
# 插入左右双引号
# 并移动到左引号之后
"\C-x\"": "\"\"\C-b"
# 插入一个反斜杠（测试序列和宏中的反斜杠转义）
"\C-x\\": "\\"
# 引用当前或前一个单词
"\C-xq": "\eb\"\ef\""
# 添加一个刷新行的绑定，该行未绑定
"\C-xr": redraw-current-line
# 编辑当前行上的变量。
"\M-\C-v": "\C-a\C-k$\C-y\M-\C-e\C-a\C-y="
$endif

# 如果有可用的可见铃声，则使用它
set bell-style visible

# 读取时不将字符剥离为7位
set input-meta on

# 允许插入iso-latin1字符而不是转换为前缀元序列
set convert-meta off

# 将设置了第八位的字符直接显示
# 而不是作为元前缀字符显示
set output-meta on

# 如果一个单词有150个或更多可能的补全项，
# 询问用户是否要查看所有补全项
set completion-query-items 150

# 对于FTP
$if Ftp
"\C-xg": "get \M-?"
"\C-xt": "put \M-?"
"\M-.": yank-last-arg
$endif
